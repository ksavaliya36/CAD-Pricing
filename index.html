<script type="text/babel">

const { useState, useRef } = React;


const METAL_PRICES = { "10K Gold": 83, "14K Gold": 110, "18K Gold": 140, "925 Silver": 10 };

const ROUND_TYPES = { "Moissanite": [10], "CVD Lab": [90], "Natural · Single Cut": [105, 155], "Natural · Full Cut": [225, 275, 325, 400] };

const FANCY_TYPES = { "Moissanite": [15], "CVD Lab": [140], "Natural (Modified Emerald)": [450] };

const MARKUP = 0.18;


const C = {

  bg: "#0B0F14", card: "#161B23", cardBorder: "rgba(255,255,255,0.06)",

  surface: "#1E242E", surface2: "#252B36",

  text: "#FFFFFF", textSoft: "#A8AEB8", textMute: "#6B7280",

  gold: "#D4A574", divider: "rgba(255,255,255,0.05)",

};


function fmt(n) { return "$" + Number(n).toFixed(0).replace(/\B(?=(\d{3})+(?!\d))/g, ","); }

function fmtDecimal(n) { return Number(n).toFixed(2); }


function Pill({ active, onClick, children, full }) {

  return (

    <button onClick={onClick} style={{

      padding: "10px 16px", borderRadius: 12, fontSize: 13, fontWeight: 500,

      border: active ? `1px solid ${C.gold}` : `1px solid transparent`,

      background: active ? "transparent" : C.surface2,

      color: active ? C.gold : C.textSoft,

      transition: "all 0.18s", flex: full ? 1 : "0 1 auto",

      textAlign: "center", whiteSpace: "nowrap",

    }}>{children}</button>

  );

}


function Card({ children, style }) {

  return <div style={{ background: C.card, border: `1px solid ${C.cardBorder}`, borderRadius: 18, padding: 18, marginBottom: 14, ...style }}>{children}</div>;

}


function FieldLabel({ children }) {

  return <p style={{ fontSize: 11, fontWeight: 600, color: C.textMute, letterSpacing: "0.12em", textTransform: "uppercase", margin: "0 0 8px" }}>{children}</p>;

}


function App() {

  const [image, setImage] = useState(null);

  const [imageB64, setImageB64] = useState(null);

  const [imageMime, setImageMime] = useState("image/jpeg");

  const [loading, setLoading] = useState(false);

  const [error, setError] = useState("");

  const [extracted, setExtracted] = useState(null);

  const [editGrams, setEditGrams] = useState("");

  const [editRound, setEditRound] = useState("");

  const [editFancy, setEditFancy] = useState("");

  const [jobName, setJobName] = useState("");

  const [jobNum, setJobNum] = useState("");

  const [apiKey, setApiKey] = useState(localStorage.getItem("anthropic_key") || "");

  const [showKeyInput, setShowKeyInput] = useState(!localStorage.getItem("anthropic_key"));

  const fileRef = useRef();


  const [metal, setMetal] = useState(null);

  const [roundType, setRoundType] = useState("Natural · Full Cut");

  const [roundPPC, setRoundPPC] = useState(225);

  const [fancyType, setFancyType] = useState(null);

  const [fancyPPC, setFancyPPC] = useState(null);


  const hasRound = parseFloat(editRound) > 0;

  const hasFancy = parseFloat(editFancy) > 0;

  const showConfig = extracted !== null;

  const roundReady = !hasRound || (roundType !== null && roundPPC !== null);

  const fancyReady = !hasFancy || (fancyType !== null && fancyPPC !== null);

  const showSummary = showConfig && metal !== null && roundReady && fancyReady;


  const metalPPG = metal ? METAL_PRICES[metal] : 0;

  const grams = parseFloat(editGrams) || 0;

  const roundCt = parseFloat(editRound) || 0;

  const fancyCt = parseFloat(editFancy) || 0;

  const subtotal = grams * metalPPG + roundCt * (roundPPC || 0) + fancyCt * (fancyPPC || 0);

  const rawFinal = subtotal * (1 + MARKUP);

  const finalPrice = rawFinal < 1500 ? Math.ceil(rawFinal / 10) * 10 : Math.ceil(rawFinal / 50) * 50;


  function handleRoundType(t) { setRoundType(t); setRoundPPC(ROUND_TYPES[t][0]); }

  function handleFancyType(t) { setFancyType(t); setFancyPPC(FANCY_TYPES[t][0]); }


  function reset() {

    setExtracted(null); setError("");

    setEditGrams(""); setEditRound(""); setEditFancy("");

    setMetal(null); setFancyType(null); setFancyPPC(null);

  }


  function saveKey() {

    localStorage.setItem("anthropic_key", apiKey);

    setShowKeyInput(false);

  }


  function handleFile(e) {

    const file = e.target.files[0]; if (!file) return;

    reset();

    setImage(URL.createObjectURL(file));

    setImageMime(file.type || "image/jpeg");

    const reader = new FileReader();

    reader.onload = () => setImageB64(reader.result.split(",")[1]);

    reader.readAsDataURL(file);

  }


  async function analyze() {

    if (!imageB64) return;

    if (!apiKey) { setShowKeyInput(true); return; }

    setLoading(true); setError(""); reset();

    try {

      const res = await fetch("https://api.anthropic.com/v1/messages", {

        method: "POST",

        headers: {

          "Content-Type": "application/json",

          "x-api-key": apiKey,

          "anthropic-version": "2023-06-01",

          "anthropic-dangerous-direct-browser-access": "true",

        },

        body: JSON.stringify({

          model: "claude-sonnet-4-5", max_tokens: 1000,

          messages: [{ role: "user", content: [

            { type: "image", source: { type: "base64", media_type: imageMime, data: imageB64 } },

            { type: "text", text: `Jewelry CAD spec sheet. Return ONLY valid JSON, no markdown:

{"grams":<number|null>,"metal_type":"<10K Gold|14K Gold|18K Gold|925 Silver|null>","round_carats":<number|null>,"round_type":"<Moissanite|CVD Lab|Natural · Single Cut|Natural · Full Cut|null>","fancy_carats":<number|null>,"fancy_type":"<Moissanite|CVD Lab|Natural (Modified Emerald)|null>","notes":"<brief summary>"}

- Gram weight: grams, wt, weight, gr, g

- Metal type: 10K/14K/18K Gold or 925/Silver/Sterling

- Round carats: RD, Round, DW, R

- Round type: Moissanite (Moiss/MOI), CVD Lab (CVD/Lab/LG), Natural · Single Cut (SC), Natural · Full Cut (FC/Brilliant/Natural)

- Fancy carats: Fancy, FD, Pear, Marquise, Princess, Oval, Cushion, Emerald

- Fancy type: Moissanite, CVD Lab, Natural (Modified Emerald)

Null if not found. Numbers only for numeric fields.` }

          ]}]

        })

      });

      if (!res.ok) {

        const errText = await res.text();

        throw new Error(`API error: ${res.status} ${errText.slice(0, 100)}`);

      }

      const data = await res.json();

      const raw = data.content.filter(b => b.type === "text").map(b => b.text).join("");

      const parsed = JSON.parse(raw.replace(/```json|```/g, "").trim());

      setExtracted(parsed);

      setEditGrams(parsed.grams != null ? String(parsed.grams) : "");

      setEditRound(parsed.round_carats != null ? String(parsed.round_carats) : "");

      setEditFancy(parsed.fancy_carats != null ? String(parsed.fancy_carats) : "");

      if (parsed.metal_type && METAL_PRICES[parsed.metal_type]) setMetal(parsed.metal_type);

      if (parsed.round_type && ROUND_TYPES[parsed.round_type]) {

        setRoundType(parsed.round_type);

        setRoundPPC(ROUND_TYPES[parsed.round_type][0]);

      }

      if (parsed.fancy_type && FANCY_TYPES[parsed.fancy_type]) {

        setFancyType(parsed.fancy_type);

        setFancyPPC(FANCY_TYPES[parsed.fancy_type][0]);

      }

    } catch (e) {

      setError(e.message || "Could not read the image. Please try again.");

    }

    setLoading(false);

  }


  const inputStyle = {

    width: "100%", padding: "13px 14px", fontSize: 15, color: C.text,

    background: C.surface, border: "none", borderRadius: 12, outline: "none", fontWeight: 500,

  };


  return (

    <div style={{ background: C.bg, padding: "20px 18px 28px", maxWidth: 480, margin: "0 auto", minHeight: "100vh" }}>


      {/* API Key prompt */}

      {showKeyInput && (

        <Card style={{ background: "#1F1813", borderColor: C.gold }}>

          <FieldLabel><span style={{color: C.gold}}>One-time setup</span></FieldLabel>

          <p style={{ fontSize: 13, color: C.textSoft, margin: "0 0 12px", lineHeight: 1.5 }}>

            Paste your Anthropic API key to enable CAD reading. It's saved on this device only.

          </p>

          <input type="password" value={apiKey} onChange={e => setApiKey(e.target.value)} placeholder="sk-ant-..." style={{...inputStyle, marginBottom: 10}} />

          <button onClick={saveKey} disabled={!apiKey} style={{

            width: "100%", padding: "13px", fontSize: 14, fontWeight: 600,

            color: "#1A1410", background: apiKey ? C.gold : C.textMute,

            border: "none", borderRadius: 12,

          }}>Save & continue</button>

          <p style={{ fontSize: 11, color: C.textMute, margin: "10px 0 0", lineHeight: 1.4 }}>

            Get a key at console.anthropic.com → API Keys

          </p>

        </Card>

      )}


      {/* Header */}

      <div style={{ display: "flex", alignItems: "flex-start", justifyContent: "space-between", marginBottom: 22, gap: 10 }}>

        <div style={{ display: "flex", gap: 12, alignItems: "flex-start" }}>

          <button onClick={() => { setImage(null); setImageB64(null); reset(); if(fileRef.current)fileRef.current.value=""; }}

            style={{ width: 38, height: 38, borderRadius: "50%", background: C.surface, border: "none", color: C.gold, fontSize: 18, display: "flex", alignItems: "center", justifyContent: "center" }}>‹</button>

          <div>

            <h1 style={{ fontSize: 26, fontWeight: 700, margin: 0, lineHeight: 1.1, letterSpacing: "-0.02em" }}>CAD<br/>Pricing</h1>

            <p style={{ fontSize: 13, color: C.textMute, margin: "4px 0 0" }}>Instant quote from CAD</p>

          </div>

        </div>

        <button onClick={() => setShowKeyInput(true)} style={{ width: 36, height: 36, borderRadius: "50%", background: C.surface, border: "none", color: C.textSoft, fontSize: 16 }}>⚙</button>

      </div>


      {/* Job info */}

      <div style={{ display: "grid", gridTemplateColumns: "1fr 110px", gap: 10, marginBottom: 14 }}>

        <div>

          <FieldLabel>Job name</FieldLabel>

          <input value={jobName} onChange={e => setJobName(e.target.value)} placeholder="e.g. Engagement Ring" style={inputStyle} />

        </div>

        <div>

          <FieldLabel>Job #</FieldLabel>

          <input value={jobNum} onChange={e => setJobNum(e.target.value)} placeholder="J1042" style={inputStyle} />

        </div>

      </div>


      {/* Upload */}

      <Card style={{ padding: image ? 14 : 18 }}>

        <FieldLabel>CAD Image</FieldLabel>

        <input ref={fileRef} type="file" accept="image/*" onChange={handleFile} style={{ display: "none" }} />

        {!image ? (

          <div onClick={() => fileRef.current.click()} style={{ borderRadius: 14, padding: "30px 16px", textAlign: "center", background: C.surface, border: `1px dashed ${C.cardBorder}` }}>

            <div style={{ fontSize: 32, color: C.gold, marginBottom: 8 }}>◈</div>

            <p style={{ fontSize: 14, fontWeight: 500, margin: "0 0 4px" }}>Tap to upload CAD image</p>

            <p style={{ fontSize: 12, color: C.textMute, margin: 0 }}>JPG, PNG, or WEBP</p>

          </div>

        ) : (

          <div>

            <img src={image} style={{ width: "100%", borderRadius: 12, maxHeight: 240, objectFit: "contain", display: "block", background: C.surface, marginBottom: 12 }} />

            {!extracted && (

              <button onClick={analyze} disabled={loading} style={{

                width: "100%", padding: "14px", fontSize: 14.5, fontWeight: 600,

                color: "#1A1410", background: loading ? C.textMute : C.gold,

                border: "none", borderRadius: 12,

              }}>{loading ? "Reading image…" : "Read CAD image"}</button>

            )}

          </div>

        )}

        {error && <p style={{ fontSize: 12.5, color: "#E89494", marginTop: 10, padding: "8px 12px", background: "rgba(232,148,148,0.1)", borderRadius: 8 }}>{error}</p>}

      </Card>


      {showConfig && (

        <>

          <Card>

            <FieldLabel>Specifications</FieldLabel>

            <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr 1fr", gap: 8 }}>

              {[["Metal", editGrams, setEditGrams, "g"], ["Round", editRound, setEditRound, "ct"], ["Fancy", editFancy, setEditFancy, "ct"]].map(([lbl, v, setV, u]) => (

                <div key={lbl}>

                  <p style={{ fontSize: 11, color: C.textMute, margin: "0 0 6px", fontWeight: 500 }}>{lbl}</p>

                  <div style={{ position: "relative" }}>

                    <input type="number" value={v} onChange={e => setV(e.target.value)} placeholder="0" step="0.01" style={{ ...inputStyle, paddingRight: 30, fontSize: 14 }} />

                    <span style={{ position: "absolute", right: 12, top: "50%", transform: "translateY(-50%)", fontSize: 11, color: C.textMute }}>{u}</span>

                  </div>

                </div>

              ))}

            </div>

          </Card>


          <Card>

            <FieldLabel>Metal</FieldLabel>

            <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 8 }}>

              {Object.keys(METAL_PRICES).map(k => <Pill key={k} active={metal === k} onClick={() => setMetal(k)} full>{k}</Pill>)}

            </div>

          </Card>


          {hasRound && (

            <Card>

              <FieldLabel>Round Diamonds</FieldLabel>

              <div style={{ display: "flex", flexWrap: "wrap", gap: 8, marginBottom: roundType && ROUND_TYPES[roundType].length > 1 ? 14 : 0 }}>

                {Object.keys(ROUND_TYPES).map(k => <Pill key={k} active={roundType === k} onClick={() => handleRoundType(k)}>{k}</Pill>)}

              </div>

              {roundType && ROUND_TYPES[roundType].length > 1 && (

                <>

                  <p style={{ fontSize: 11, color: C.textMute, margin: "0 0 8px", fontWeight: 500 }}>Quality grade</p>

                  <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 8 }}>

                    {ROUND_TYPES[roundType].map((p, i) => <Pill key={p} active={roundPPC === p} onClick={() => setRoundPPC(p)} full>{["Standard", "Premium", "Superior", "Exceptional"][i]}</Pill>)}

                  </div>

                </>

              )}

            </Card>

          )}


          {hasFancy && (

            <Card>

              <FieldLabel>Fancy Diamonds</FieldLabel>

              <div style={{ display: "flex", flexWrap: "wrap", gap: 8 }}>

                {Object.keys(FANCY_TYPES).map(k => <Pill key={k} active={fancyType === k} onClick={() => handleFancyType(k)}>{k}</Pill>)}

              </div>

            </Card>

          )}

        </>

      )}


      {showSummary && (

        <Card style={{ padding: 0, overflow: "hidden", marginTop: 18 }}>

          <div style={{ padding: "16px 18px", display: "flex", alignItems: "center", gap: 8 }}>

            <span style={{ color: C.gold, fontSize: 14 }}>◆</span>

            <FieldLabel><span style={{ color: C.gold }}>Quote Summary</span></FieldLabel>

          </div>

          <div style={{ padding: "0 18px" }}>

            <div style={{ display: "flex", justifyContent: "space-between", padding: "12px 0", borderTop: `1px solid ${C.divider}` }}>

              <span style={{ color: C.textSoft, fontSize: 14 }}>Metal</span>

              <span style={{ fontSize: 14, fontWeight: 500 }}>{metal} · {fmtDecimal(grams)}g</span>

            </div>

            {hasRound && (

              <div style={{ display: "flex", justifyContent: "space-between", padding: "12px 0", borderTop: `1px solid ${C.divider}` }}>

                <span style={{ color: C.textSoft, fontSize: 14 }}>Round</span>

                <span style={{ fontSize: 14, fontWeight: 500, textAlign: "right" }}>{roundType.replace("Natural · ", "")} · {fmtDecimal(roundCt)}ct</span>

              </div>

            )}

            {hasFancy && (

              <div style={{ display: "flex", justifyContent: "space-between", padding: "12px 0", borderTop: `1px solid ${C.divider}` }}>

                <span style={{ color: C.textSoft, fontSize: 14 }}>Fancy</span>

                <span style={{ fontSize: 14, fontWeight: 500, textAlign: "right" }}>{fancyType.replace(" (Modified Emerald)", "")} · {fmtDecimal(fancyCt)}ct</span>

              </div>

            )}

          </div>

          <div style={{ background: C.surface, padding: "20px 18px", marginTop: 4 }}>

            <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>

              <span style={{ color: C.gold, fontSize: 14, fontWeight: 600 }}>Grand Total</span>

              <span style={{ color: C.gold, fontSize: 32, fontWeight: 700, letterSpacing: "-0.02em" }}>{fmt(finalPrice)}</span>

            </div>

            {(jobName || jobNum) && (

              <p style={{ fontSize: 11.5, color: C.textMute, margin: "4px 0 0", textAlign: "right" }}>

                {jobName}{jobName && jobNum ? " · " : ""}{jobNum && `#${jobNum}`}

              </p>

            )}

          </div>

          <div style={{ padding: 14, background: C.surface }}>

            <button onClick={() => {

              const text = `Quote\n${jobName ? jobName + " " : ""}${jobNum ? "#" + jobNum : ""}\n\n${metal} · ${fmtDecimal(grams)}g${hasRound ? `\n${roundType} · ${fmtDecimal(roundCt)}ct rounds` : ""}${hasFancy ? `\n${fancyType} · ${fmtDecimal(fancyCt)}ct fancy` : ""}\n\nTotal: ${fmt(finalPrice)}`;

              if (navigator.share) navigator.share({ text }); else { navigator.clipboard.writeText(text); alert("Copied to clipboard!"); }

            }} style={{

              width: "100%", padding: "15px", fontSize: 15, fontWeight: 600,

              color: "#1A1410", background: C.gold, border: "none", borderRadius: 14,

              display: "flex", alignItems: "center", justifyContent: "center", gap: 8,

            }}><span style={{ fontSize: 16 }}>↑</span> Export Quote</button>

          </div>

        </Card>

      )}


      <p style={{ fontSize: 11, color: C.textMute, textAlign: "center", margin: "20px 0 0", letterSpacing: "0.05em" }}>

        Prime Wholesale · Quote valid for 7 days

      </p>

    </div>

  );

}


ReactDOM.createRoot(document.getElementById("root")).render(<App />);

</script>

</body>

</html>
